var whatTimeIsIt = function(angle) {
  let timeInMinutes = angle * 720 / 360;
  let hour = Math.floor(timeInMinutes / 60);
  let minutes = Math.floor(timeInMinutes % 60);
  if (hour == 0)
    hour = 12;
  if (hour <= 9)
    hour = String('0' + hour);
  if (minutes <= 9)
    minutes = String('0' + minutes);
